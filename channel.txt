Run following commands for all 3 buff QMs (ALPHA_MASTERBUFF, BRAVO_MASTERBUFF, CHARLIE_MASTERBUFF)-

SET CHLAUTH(SVRCON) TYPE(BLOCKUSER) USERLIST('nobody')
SET CHLAUTH(SVRCON) TYPE(ADDRESSMAP) ADDRESS(*) USERSRC(NOACCESS)
SET CHLAUTH(SVRCON) TYPE(ADDRESSMAP) ADDRESS('*')  USERSRC(CHANNEL)
REFRESH SECURITY TYPE(AUTHSERV)

ALTER AUTHINFO(SYSTEM.DEFAULT.AUTHINFO.IDPWOS) AUTHTYPE(IDPWOS) CHCKCLNT(NONE) CHCKLOCL(NONE)
ALTER CHL(SVRCON) CHLTYPE(SVR) MCAUSER('NOAUTH')

STOP CHL(SVRCON)
START CHL(SVRCON)
